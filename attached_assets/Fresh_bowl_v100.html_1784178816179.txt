    <script>
        // ════════════════════════════════════════════════════════════════
        //  DATA LAYER
        // ════════════════════════════════════════════════════════════════

        const STORAGE_KEY = 'dog_health_calculators_data';

        function getDefaultData() {
            return {
                dogs: {},
                activeDogId: null,
                nextId: 1
            };
        }

        function loadData() {
            try {
                const raw = localStorage.getItem(STORAGE_KEY);
                if (raw) {
                    const data = JSON.parse(raw);
                    // ensure structure
                    if (!data.dogs) data.dogs = {};
                    if (!data.nextId) data.nextId = 1;
                    return data;
                }
            } catch (_) { /* ignore */ }
            return getDefaultData();
        }

        function saveData(data) {
            try {
                localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
            } catch (_) { /* ignore */ }
        }

        function getActiveDog(data) {
            if (!data.activeDogId) return null;
            return data.dogs[data.activeDogId] || null;
        }

        function getDogName(data, id) {
            const d = data.dogs[id];
            return d ? d.name || 'Unnamed' : 'Unknown';
        }

        function ensureDogRecords(data, dogId) {
            if (!data.dogs[dogId]) return;
            if (!data.dogs[dogId].records) data.dogs[dogId].records = {};
            const rec = data.dogs[dogId].records;
            const types = ['rr', 'calories', 'chocolate', 'flea', 'exercise', 'treats', 'cost'];
            types.forEach(t => { if (!rec[t]) rec[t] = []; });
        }

        function pushRecord(data, dogId, type, entry) {
            if (!data.dogs[dogId]) return;
            ensureDogRecords(data, dogId);
            entry.date = new Date().toISOString();
            data.dogs[dogId].records[type].push(entry);
            saveData(data);
        }

        function getRecords(data, dogId, type) {
            if (!data.dogs[dogId]) return [];
            ensureDogRecords(data, dogId);
            return data.dogs[dogId].records[type] || [];
        }

        function getAllRecords(data, dogId) {
            if (!data.dogs[dogId]) return [];
            ensureDogRecords(data, dogId);
            const rec = data.dogs[dogId].records;
            const flat = [];
            const labels = {
                rr: '💨 RR',
                calories: '🍽️ Calories',
                chocolate: '🍫 Chocolate',
                flea: '🐜 Flea',
                exercise: '🏃 Exercise',
                treats: '🍪 Treats',
                cost: '💰 Cost'
            };
            Object.keys(rec).forEach(type => {
                rec[type].forEach(entry => {
                    flat.push({ type: labels[type] || type, ...entry });
                });
            });
            flat.sort((a, b) => new Date(b.date) - new Date(a.date));
            return flat;
        }

        // ── Dog management ──

        function addDog(name) {
            const data = loadData();
            const id = 'dog_' + data.nextId++;
            data.dogs[id] = { name: name || 'Unnamed', records: {} };
            data.activeDogId = id;
            saveData(data);
            renderAll();
            return id;
        }

        function deleteDog(id) {
            const data = loadData();
            if (!data.dogs[id]) return;
            if (!confirm('Delete "' + (data.dogs[id].name || 'Unnamed') + '" and all their records?')) return;
            delete data.dogs[id];
            if (data.activeDogId === id) {
                const keys = Object.keys(data.dogs);
                data.activeDogId = keys.length ? keys[0] : null;
            }
            saveData(data);
            renderAll();
        }

        function switchDog(id) {
            const data = loadData();
            if (data.dogs[id]) {
                data.activeDogId = id;
                saveData(data);
                renderAll();
            }
        }

        // ── Render helpers ──

        let _pendingSave = null; // { type, data }

        function renderAll() {
            renderDogBar();
            renderTabs();
            renderRecords();
            // Re-run any visible panel's display
            const active = document.querySelector('.panel.active');
            if (active) {
                const id = active.id.replace('panel-', '');
                // refresh history for that panel
                renderHistory(id);
            }
            // update save prompt dog names
            document.querySelectorAll('.save-prompt [id$="SaveDogName"]').forEach(el => {
                const data = loadData();
                const dog = getActiveDog(data);
                el.textContent = dog ? dog.name || 'Unnamed' : '—';
            });
        }

        function renderDogBar() {
            const data = loadData();
            const bar = document.getElementById('dogBar');
            const activeId = data.activeDogId;
            let html = '';
            Object.keys(data.dogs).forEach(id => {
                const d = data.dogs[id];
                const active = id === activeId ? 'active' : '';
                html += `<span class="dog-chip ${active}">
                            🐕 ${d.name || 'Unnamed'}
                            <span class="del" onclick="event.stopPropagation();deleteDog('${id}')">✕</span>
                         </span>`;
            });
            html += `<span class="dog-chip add" onclick="promptAddDog()">＋ Add Dog</span>`;
            bar.innerHTML = html;
            // click handlers for switching
            bar.querySelectorAll('.dog-chip:not(.add)').forEach(el => {
                const id = el.textContent.replace(/✕$/, '').trim();
                // find the actual id from data
                const data2 = loadData();
                Object.keys(data2.dogs).forEach(did => {
                    if (el.textContent.includes(data2.dogs[did].name || 'Unnamed')) {
                        el.onclick = () => switchDog(did);
                    }
                });
            });
        }

        function promptAddDog() {
            const name = prompt('Enter your dog\'s name:');
            if (name !== null && name.trim()) {
                addDog(name.trim());
            } else if (name !== null) {
                addDog('Unnamed');
            }
        }

        function renderTabs() {
            // tabs are static, just ensure active class
            const activePanel = document.querySelector('.panel.active');
            if (activePanel) {
                const id = activePanel.id.replace('panel-', '');
                document.querySelectorAll('#tabNav button').forEach(btn => {
                    btn.classList.toggle('active', btn.dataset.tab === id);
                });
            }
        }

        // ── Tab switching ──

        document.getElementById('tabNav').addEventListener('click', (e) => {
            const btn = e.target.closest('button');
            if (!btn) return;
            const tab = btn.dataset.tab;
            document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));
            document.querySelectorAll('#tabNav button').forEach(b => b.classList.remove('active'));
            document.getElementById('panel-' + tab).classList.add('active');
            btn.classList.add('active');
            renderHistory(tab);
        });

        // ── History render ──

        function renderHistory(type) {
            const data = loadData();
            const dog = getActiveDog(data);
            if (!dog) {
                document.getElementById(type + 'History').innerHTML =
                    '<div class="empty-records">No dog selected. Add a dog first!</div>';
                return;
            }
            const records = getRecords(data, data.activeDogId, type);
            const container = document.getElementById(type + 'History');
            if (!records.length) {
                container.innerHTML = '<div class="empty-records">No records saved yet.</div>';
                return;
            }
            let html = '';
            records.slice().reverse().forEach(r => {
                const dateStr = new Date(r.date).toLocaleDateString() + ' ' +
                    new Date(r.date).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
                html += `<div class="record-item">
                            <div class="date">${dateStr}</div>
                            <div class="data">${formatRecord(type, r)}</div>
                         </div>`;
            });
            container.innerHTML = html;
        }

        function formatRecord(type, r) {
            switch (type) {
                case 'rr':
                    return `💨 <strong>${r.bpm}</strong> BPM (${r.window}s window, ${r.taps} taps) — ${r.status || 'normal'}`;
                case 'calories':
                    return `🍽️ <strong>${Math.round(r.mer)}</strong> kcal/day, ${r.meals} meal${r.meals>1?'s':''}, ~${Math.round(r.ozPerMeal||0)} oz/meal`;
                case 'chocolate':
                    return `🍫 <strong>${r.dose}</strong> mg/kg — ${r.risk} (${r.chocType}, ${r.amount}g)`;
                case 'flea':
                    return `🐜 ${r.medication || 'Medication'} — next dose: ${r.nextDates ? r.nextDates[0] : '—'}`;
                case 'exercise':
                    return `🏃 ${r.plan} — ${r.detail || ''}`;
                case 'treats':
                    return `🍪 <strong>${Math.round(r.budget)}</strong> kcal/day treat budget (${Math.round(r.pctOfDaily)}% of daily)`;
                case 'cost':
                    return `💰 ${r.foodA || 'Food A'}: $${r.costA?.toFixed(2)||'—'}/day ${r.foodB ? 'vs '+r.foodB+': $'+r.costB?.toFixed(2) : ''}`;
                default:
                    return JSON.stringify(r);
            }
        }

        function renderRecords() {
            const data = loadData();
            const dog = getActiveDog(data);
            const container = document.getElementById('allRecords');
            if (!dog) {
                container.innerHTML = '<div class="empty-records">No dog selected. Add a dog first!</div>';
                return;
            }
            const all = getAllRecords(data, data.activeDogId);
            if (!all.length) {
                container.innerHTML = '<div class="empty-records">No records saved yet. Run a calculator and save the result!</div>';
                return;
            }
            let html = '';
            all.forEach(r => {
                const dateStr = new Date(r.date).toLocaleDateString() + ' ' +
                    new Date(r.date).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
                const label = r.type || '📋';
                const summary = formatRecordFromEntry(r);
                html += `<div class="record-item">
                            <div class="date">${dateStr} ${label}</div>
                            <div class="data">${summary}</div>
                         </div>`;
            });
            container.innerHTML = html;
        }

        function formatRecordFromEntry(r) {
            // generic formatter for the all-records view
            if (r.bpm) return `💨 <strong>${r.bpm}</strong> BPM`;
            if (r.mer) return `🍽️ <strong>${Math.round(r.mer)}</strong> kcal/day`;
            if (r.dose) return `🍫 <strong>${r.dose}</strong> mg/kg — ${r.risk}`;
            if (r.nextDates) return `🐜 next: ${r.nextDates[0]}`;
            if (r.plan) return `🏃 ${r.plan}`;
            if (r.budget) return `🍪 <strong>${Math.round(r.budget)}</strong> kcal`;
            if (r.costA) return `💰 $${r.costA.toFixed(2)}/day`;
            return '📋 ' + JSON.stringify(r).slice(0, 60);
        }

        // ════════════════════════════════════════════════════════════════
        //  1. RESPIRATORY RATE
        // ════════════════════════════════════════════════════════════════

        let rrState = { taps: 0, running: false, timer: null, seconds: 0, window: 30, bpm: null };

        document.getElementById('rrWindow').addEventListener('change', function() {
            rrState.window = parseInt(this.value) || 30;
            resetRR();
        });

        document.getElementById('rrTapBtn').addEventListener('click', function() {
            if (!rrState.running) {
                startRR();
                return;
            }
            rrState.taps++;
            document.getElementById('rrTaps').textContent = 'Taps: ' + rrState.taps;
            // haptic feedback if available
            if (navigator.vibrate) navigator.vibrate(10);
        });

        function startRR() {
            resetRR();
            rrState.running = true;
            rrState.seconds = 0;
            rrState.taps = 0;
            const btn = document.getElementById('rrTapBtn');
            btn.textContent = '👆 Tap for each breath';
            btn.classList.add('timer-active');
            document.getElementById('rrTaps').textContent = 'Taps: 0';
            document.getElementById('rrTimer').textContent = '⏱️ 0s';
            document.getElementById('rrResult').style.display = 'none';
            document.getElementById('rrStatus').textContent = '⏳ Counting… tap each breath.';
            rrState.timer = setInterval(() => {
                rrState.seconds++;
                document.getElementById('rrTimer').textContent = '⏱️ ' + rrState.seconds + 's';
                if (rrState.seconds >= rrState.window) {
                    stopRR();
                }
            }, 1000);
        }

        function stopRR() {
            rrState.running = false;
            if (rrState.timer) { clearInterval(rrState.timer);
                rrState.timer = null; }
            const btn = document.getElementById('rrTapBtn');
            btn.classList.remove('timer-active');
            btn.textContent = '👆 Tap for each breath';
            const taps = rrState.taps;
            const windowSec = rrState.window;
            let bpm = 0;
            let status = 'normal';
            let statusText = 'Normal sleeping rate.';
            if (taps > 0) {
                bpm = Math.round((taps / windowSec) * 60);
                if (bpm < 30) status = 'normal';
                else if (bpm < 40) { status = 'borderline';
                    statusText = 'Borderline — re-measure in a cool, quiet room. If sustained, contact your vet.'; } else { status =
                        'high';
                    statusText = '⚠️ Abnormal sleeping rate — contact your vet, especially if your dog has a heart condition.'; }
            }
            rrState.bpm = bpm;
            document.getElementById('rrStatus').textContent = taps === 0 ? 'No taps recorded. Try again.' : '✅ Done!';
            if (taps > 0) {
                const result = document.getElementById('rrResult');
                result.style.display = 'block';
                document.getElementById('rrResultBpm').textContent = bpm + ' BPM';
                document.getElementById('rrResultDetail').innerHTML =
                    `${taps} taps in ${windowSec}s &bull; ${statusText}` +
                    (status === 'normal' ? ' ✅ Healthy' : '') +
                    (status === 'borderline' ? ' ⚠️ Watch' : '') +
                    (status === 'high' ? ' 🚨 Alert' : '');
                // Store for save
                _pendingSave = {
                    type: 'rr',
                    data: { bpm, taps, window: windowSec, status, statusText }
                };
                showSavePrompt('rr');
            } else {
                dismissSave('rr');
            }
            document.getElementById('rrTaps').textContent = 'Taps: ' + taps;
        }

        function resetRR() {
            rrState.running = false;
            if (rrState.timer) { clearInterval(rrState.timer);
                rrState.timer = null; }
            rrState.taps = 0;
            rrState.seconds = 0;
            rrState.bpm = null;
            const btn = document.getElementById('rrTapBtn');
            btn.classList.remove('timer-active');
            btn.textContent = '👆 Tap for each breath';
            document.getElementById('rrTaps').textContent = 'Taps: 0';
            document.getElementById('rrTimer').textContent = '⏱️ 0s';
            document.getElementById('rrStatus').textContent = '';
            document.getElementById('rrResult').style.display = 'none';
            dismissSave('rr');
            _pendingSave = null;
        }

        function saveRR() {
            const data = loadData();
            const dog = getActiveDog(data);
            if (!dog) { alert('No dog selected.'); return; }
            if (!_pendingSave || _pendingSave.type !== 'rr') return;
            pushRecord(data, data.activeDogId, 'rr', _pendingSave.data);
            dismissSave('rr');
            _pendingSave = null;
            renderAll();
            renderHistory('rr');
            // show a quick feedback
            const status = document.getElementById('rrStatus');
            status.textContent = '✅ Saved to ' + (dog.name || 'Unnamed') + '\'s profile!';

        // ════════════════════════════════════════════════════════════════
        //  2. CALORIES
        // ════════════════════════════════════════════════════════════════

        function calcCalories() {
            const weight = parseFloat(document.getElementById('calWeight').value) || 25;
            const age = document.getElementById('calAge').value;
            const activity = parseFloat(document.getElementById('calActivity').value) || 1.6;
            const condition = parseFloat(document.getElementById('calCondition').value) || 1.0;
            const neutered = parseFloat(document.getElementById('calNeutered').value) || 1.0;
            const meals = parseInt(document.getElementById('calMeals').value) || 2;
            const treatPct = parseFloat(document.getElementById('calTreats').value) || 0;

            let ageFactor = 1.0;
            if (age === 'puppy_under4') ageFactor = 3.0;
            else if (age === 'puppy_4to12') ageFactor = 2.0;
            else if (age === 'senior') ageFactor = 0.9;

            const rer = 70 * Math.pow(weight, 0.75);
            let mer = rer * activity * condition * neutered * ageFactor;
            const treatShare = treatPct / 100;
            const foodCalories = mer * (1 - treatShare);
            const ozPerDay = foodCalories / 42; // approximate density
            const ozPerMeal = ozPerDay / meals;

            const result = {
                weight,
                age,
                activity,
                condition,
                neutered,
                meals,
                treatPct,
                rer: Math.round(rer),
                mer: Math.round(mer),
                foodCalories: Math.round(foodCalories),
                ozPerDay: Math.round(ozPerDay * 10) / 10,
                ozPerMeal: Math.round(ozPerMeal * 10) / 10,
                treatKcal: Math.round(mer * treatShare)
            };

            document.getElementById('calResult').style.display = 'block';
            document.getElementById('calMer').textContent = Math.round(mer) + ' kcal/day';
            document.getElementById('calDetail').innerHTML =
                `RER: <strong>${Math.round(rer)}</strong> kcal &bull; ` +
                `Food: <strong>${Math.round(foodCalories)}</strong> kcal/day &bull; ` +
                `~<strong>${result.ozPerDay}</strong> oz/day (${result.ozPerMeal} oz/meal) &bull; ` +
                `Treats: <strong>${result.treatKcal}</strong> kcal/day`;

            _pendingSave = { type: 'calories', data: result };
            showSavePrompt('cal');
        }

        function saveCalories() {
            const data = loadData();
            const dog = getActiveDog(data);
            if (!dog) { alert('No dog selected.'); return; }
            if (!_pendingSave || _pendingSave.type !== 'calories') return;
            pushRecord(data, data.activeDogId, 'calories', _pendingSave.data);
            dismissSave('cal');
            _pendingSave = null;
            renderAll();
            renderHistory('calories');
        }

        //  3. CHOCOLATE TOXICITY
        // ════════════════════════════════════════════════════════════════

        function calcChocolate() {
            const weight = parseFloat(document.getElementById('chocWeight').value) || 25;
            const chocType = parseInt(document.getElementById('chocType').value) || 150;
            const amount = parseFloat(document.getElementById('chocAmount').value) || 0;

            if (weight <= 0 || amount <= 0) {
                alert('Please enter a valid weight and amount.');
                return;
            }

            const theobromineMg = amount * (chocType / 1000);
            const dose = theobromineMg / weight;
            let risk = 'Low';
            let color = 'badge-safe';
            let advice = 'No immediate concern. Monitor for any unusual behavior.';
            if (dose > 200) { risk = 'Extreme';
                color = 'badge-danger';
                advice = '🚨 EMERGENCY — call your vet or poison control immediately!'; } else if (dose > 100) { risk = 'High';
                color = 'badge-danger';
                advice = '⚠️ High risk — contact your vet right away.'; } else if (dose > 40) { risk = 'Moderate';
                color = 'badge-warn';
                advice = '⚠️ Moderate risk — call your vet for guidance.'; } else if (dose > 10) { risk = 'Mild';
                color = 'badge-warn';
                advice = 'Watch for symptoms like vomiting, diarrhea, restlessness.'; } else { risk = 'Low';
                color = 'badge-safe';
                advice = 'Likely safe, but monitor your dog just in case.'; }

            const result = {
                weight,
                chocType,
                amount,
                dose: Math.round(dose * 100) / 100,
                risk,
                color,
                advice,
                chocLabel: document.getElementById('chocType').selectedOptions[0].text
            };

            document.getElementById('chocResult').style.display = 'block';
            document.getElementById('chocDose').textContent = result.dose + ' mg/kg';
            document.getElementById('chocDetail').innerHTML =
                `<span class="badge ${color}">${risk}</span> &bull; ${amount}g ${result.chocLabel} &bull; ${advice}`;

            _pendingSave = { type: 'chocolate', data: result };
            showSavePrompt('choc');
        }

        function calcFlea() {
            const med = document.getElementById('fleaMed').value.trim() || 'Medication';
            const lastDate = document.getElementById('fleaLastDate').value;
            const freq = parseInt(document.getElementById('fleaFreq').value) || 30;

            if (!lastDate) {
                alert('Please select the last dose date.');
                return;
            }

            const dates = [];
            let d = new Date(lastDate + 'T00:00:00');
            for (let i = 0; i < 6; i++) {
                d.setDate(d.getDate() + freq);
                dates.push(d.toISOString().split('T')[0]);
            }

            const result = {
                medication: med,
                lastDate,
                frequency: freq,
                nextDates: dates
            };

            const container = document.getElementById('fleaScheduleList');
            container.innerHTML = dates.map((dt, i) =>
                `<div class="item"><span class="num">#${i+1}</span><span>${dt}</span></div>`
            ).join('');

            document.getElementById('fleaResult').style.display = 'block';
            _pendingSave = { type: 'flea', data: result };

        // ════════════════════════════════════════════════════════════════

        function calcExercise() {
            const breed = document.getElementById('exBreed').value;
            const age = document.getElementById('exAge').value;
            const weight = parseFloat(document.getElementById('exWeight').value) || 25;
            const health = document.getElementById('exHealth').value;

            const breedMap = {
                toy: { walk: 20, play: 10 },
                small: { walk: 30, play: 15 },
                medium: { walk: 45, play: 20 },
                large: { walk: 60, play: 25 },
                giant: { walk: 50, play: 15 }
            };
            let base = breedMap[breed] || breedMap.medium;
            let walk = base.walk;
            let play = base.play;

            if (age === 'puppy') { walk = Math.round(walk * 0.6);
                play = Math.round(play * 0.5); } else if (age === 'senior') { walk = Math.round(walk * 0.7);
                play = Math.round(play * 0.5); }

            if (health === 'brachy') { walk = Math.round(walk * 0.6);
                play = Math.round(play * 0.5); } else if (health === 'cardiac') { walk = Math.round(walk * 0.4);
                play = Math.round(play * 0.3); } else if (health === 'ortho') { walk = Math.round(walk * 0.5);
                play = Math.round(play * 0.4); }

            const plan = `${walk} min walk + ${play} min play`;
            let detail = `For a ${weight}kg ${age} dog, ${health.replace('_',' ')}. `;
            if (health !== 'healthy') detail += '⚠️ Adjust to your dog\'s tolerance and consult your vet.';

            const result = { breed, age, weight, health, walk, play, plan, detail };

            document.getElementById('exResult').style.display = 'block';
            document.getElementById('exPlan').textContent = plan;
            document.getElementById('exDetail').textContent = detail;

            _pendingSave = { type: 'exercise', data: result };
            showSavePrompt('ex');
        }

        //  7. TRUE COST PER DAY
        // ════════════════════════════════════════════════════════════════

        function calcCost() {
            const nameA = document.getElementById('costNameA').value.trim() || 'Food A';
            const priceA = parseFloat(document.getElementById('costPriceA').value) || 0;
            const sizeA = parseFloat(document.getElementById('costSizeA').value) || 1;
            const portionA = parseFloat(document.getElementById('costPortionA').value) || 100;

            const nameB = document.getElementById('costNameB').value.trim() || 'Food B';
            const priceB = parseFloat(document.getElementById('costPriceB').value) || 0;
            const sizeB = parseFloat(document.getElementById('costSizeB').value) || 1;
            const portionB = parseFloat(document.getElementById('costPortionB').value) || 100;

            if (priceA <= 0 || sizeA <= 0 || portionA <= 0) {
                alert('Please fill in Food A completely.');
                return;
            }

            const daysA = sizeA * 1000 / portionA;
            const costA = priceA / daysA;
            const monthA = costA * 30;
            const yearA = costA * 365;

            let result = {
                foodA: nameA,
                foodB: nameB,
                priceA,
                sizeA,
                portionA,
                daysA: Math.round(daysA * 10) / 10,
                costA: Math.round(costA * 100) / 100,
                monthA: Math.round(monthA * 100) / 100,
                yearA: Math.round(yearA * 100) / 100,
            };

            let detailHtml =
                `<strong>${nameA}</strong>: $${result.costA}/day · $${result.monthA}/mo · $${result.yearA}/yr · ${result.daysA} days/bag`;

            if (priceB > 0 && sizeB > 0 && portionB > 0) {
                const daysB = sizeB * 1000 / portionB;
                const costB = priceB / daysB;
                const monthB = costB * 30;
                const yearB = costB * 365;
                result.costB = Math.round(costB * 100) / 100;
                result.monthB = Math.round(monthB * 100) / 100;
                result.yearB = Math.round(yearB * 100) / 100;
                result.daysB = Math.round(daysB * 10) / 10;
                result.priceB = priceB;
                result.sizeB = sizeB;
                result.portionB = portionB;

                const cheaper = costA < costB ? nameA : nameB;
                const diff = Math.abs(costA - costB);
                detailHtml +=
                    `<br><strong>${nameB}</strong>: $${result.costB}/day · $${result.monthB}/mo · $${result.yearB}/yr · ${result.daysB} days/bag`;
                detailHtml += `<br>🏆 <strong>${cheaper}</strong> is cheaper by <strong>$${diff.toFixed(2)}</strong> per day`;
            }

            document.getElementById('costResult').style.display = 'block';
            document.getElementById('costDetail').innerHTML = detailHtml;

            _pendingSave = { type: 'cost', data: result };
            showSavePrompt('cost');
        }

        function saveCost() {
            const data = loadData();
            const dog = getActiveDog(data);
            if (!dog) { alert('No dog selected.'); return; }
            if (!_pendingSave || _pendingSave.type !== 'cost') return;
            pushRecord(data, data.activeDogId, 'cost', _pendingSave.data);

            renderHistory('cost');
        }
